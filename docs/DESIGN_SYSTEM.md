const POOL_COLORS = {
  match: process.env.NEXT_PUBLIC_COLOR_MATCH || '#22c55e',
  noMatch: process.env.NEXT_PUBLIC_COLOR_NO_MATCH || '#a855f7',
};

// –ü–∞—Ä—Å–∏–Ω–≥ hex –≤ RGB
function hexToRgb(hex: string): { r: number; g: number; b: number } {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? {
    r: parseInt(result[1], 16),
    g: parseInt(result[2], 16),
    b: parseInt(result[3], 16),
  } : { r: 0, g: 0, b: 0 };
}

// –ò–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è –º–µ–∂–¥—É —Ü–≤–µ—Ç–∞–º–∏ –ø–æ score
function interpolateColor(score: number): string {
  const from = hexToRgb(POOL_COLORS.noMatch); // score=0
  const to = hexToRgb(POOL_COLORS.match);     // score=1
  
  const r = Math.round(from.r + (to.r - from.r) * score);
  const g = Math.round(from.g + (to.g - from.g) * score);
  const b = Math.round(from.b + (to.b - from.b) * score);
  
  return `rgb(${r}, ${g}, ${b})`;
}
